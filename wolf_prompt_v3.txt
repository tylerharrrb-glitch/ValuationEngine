7.2 — Reverse DCF
Display panel: 'What terminal growth rate does the current market price imply?' Solve numerically:
function solveImpliedGrowth(marketPrice, shares, totalDebt, cash, sumPvFcff, fcffN, wacc, n) {
  const impliedEV = marketPrice * shares + totalDebt - cash;
  let gLow = 0, gHigh = wacc - 0.001, tol = 0.0001;
  while (gHigh - gLow > tol) {
    const gMid = (gLow + gHigh) / 2;
    const tv   = fcffN * (1 + gMid) / (wacc - gMid);
    const pvTv = tv / Math.pow(1 + wacc, n);
    (sumPvFcff + pvTv > impliedEV) ? gHigh = gMid : gLow = gMid;
  }
  return (gLow + gHigh) / 2;
}
Display: implied g, user's base case g, interpretation ('Market is MORE/LESS optimistic than your model').
